<script>
// ────────────────────────────────────────────────
let grosh = 0;
let cps   = 0;
let username = "Player" + Math.floor(Math.random() * 9000 + 1000);

// Format large numbers (k, M, B, T...)
function format(n) {
  if (n < 1e3) return Math.floor(n);
  const units = ["", "k", "M", "B", "T", "Qa", "Qi"];
  let u = 0;
  while (n >= 1e3 && u < units.length - 1) { n /= 1e3; u++; }
  return n.toFixed(n < 10 ? 1 : 0) + units[u];
}

// ────────────────────────────────────────────────
const skillTree = {
  clickBoost: { cost:  10, owned: false, desc: "+1 per click"       },
  cpsBoost:   { cost:  25, owned: false, desc: "+5 CPS"             },
  megaBoost:  { cost: 120, owned: false, desc: "×2 CPS multiplier"  },
};

const upgrades = {
  Cursor:   { baseCost:  50, cost:  50, power: 1,   count: 0 },
  Factory:  { baseCost: 220, cost: 220, power: 4.5, count: 0 },
  // you can easily add more later
};

function updateUI() {
  document.getElementById("groshDisplay").innerText = format(grosh) + " Grosh";
  document.getElementById("cps").innerText = format(cps);
}

function clickGrosh(e) {
  const add = 1 + (skillTree.clickBoost.owned ? 1 : 0);
  grosh += add;

  spawnParticles(e.clientX, e.clientY);
  spawnFloatText(e.clientX, e.clientY, "+" + format(add));

  updateUI();
}

// ─── Particles ──────────────────────────────────────
function spawnParticles(x, y) {
  for (let i = 0; i < 18; i++) {
    const p = document.createElement("div");
    p.className = "particle";
    const angle = Math.random() * Math.PI * 2;
    const dist = 40 + Math.random() * 120;
    const dx = Math.cos(angle) * dist + "px";
    const dy = Math.sin(angle) * dist + "px";
    p.style.setProperty("--dx", dx);
    p.style.setProperty("--dy", dy);
    p.style.left = x + "px";
    p.style.top  = y + "px";
    document.body.appendChild(p);
    setTimeout(() => p.remove(), 1200);
  }
}

function spawnFloatText(x, y, txt) {
  const t = document.createElement("div");
  t.className = "float-text";
  t.innerText = txt;
  t.style.left = x + "px";
  t.style.top  = y + "px";
  document.body.appendChild(t);
  setTimeout(() => t.remove(), 1400);
}

// ─── Passive income (delta-time aware) ──────────────
let lastTime = performance.now();
function gameLoop(now) {
  const dt = (now - lastTime) / 1000;
  lastTime = now;

  grosh += cps * dt;
  updateUI();

  requestAnimationFrame(gameLoop);
}
requestAnimationFrame(gameLoop);

// ─── Shop ───────────────────────────────────────────
function renderShop() {
  const s = document.getElementById("shopItems");
  s.innerHTML = "";
  for (const [key, u] of Object.entries(upgrades)) {
    const div = document.createElement("div");
    div.className = "panel-item";
    div.innerHTML = `
      ${key} (${u.count})<br>
      +${format(u.power)} CPS<br>
      Cost: ${format(u.cost)}
    `;
    div.onclick = () => buyUpgrade(key);
    if (grosh < u.cost) div.classList.add("locked");
    s.appendChild(div);
  }
}

function buyUpgrade(key) {
  const u = upgrades[key];
  if (grosh >= u.cost) {
    grosh -= u.cost;
    cps += u.power;
    u.count++;
    u.cost = Math.floor(u.baseCost * Math.pow(1.15, u.count));
    renderShop();
    updateUI();
  }
}

// ─── Skills ─────────────────────────────────────────
function renderSkills() {
  const s = document.getElementById("skills");
  s.innerHTML = "";
  for (const [key, sk] of Object.entries(skillTree)) {
    const div = document.createElement("div");
    div.className = "panel-item";
    if (sk.owned) {
      div.innerHTML = `${sk.desc} <small>(owned)</small>`;
      div.classList.add("locked");
    } else {
      div.innerHTML = `${sk.desc}<br>Cost: ${format(sk.cost)}`;
      div.onclick = () => buySkill(key);
      if (grosh < sk.cost) div.classList.add("locked");
    }
    s.appendChild(div);
  }
}

function buySkill(key) {
  const sk = skillTree[key];
  if (!sk.owned && grosh >= sk.cost) {
    grosh -= sk.cost;
    sk.owned = true;

    if (key === "clickBoost") { /* already handled in click */ }
    if (key === "cpsBoost")    cps += 5;
    if (key === "megaBoost")   cps *= 2;

    renderSkills();
    updateUI();
  }
}

function toggleSpace() {
  document.body.classList.toggle("space");
}

  `;
}
